#include <LiquidCrystal.h> //lcd ekran Kütüphanesi
LiquidCrystal lcd(12,11,5,4,3,2);//Lcd ekran baðlantý pinleri

const int trig =9; //Hcr-04 sensör trig pini
const int echo =8; //Hcr-04 sensör echo pini
const int LED =13; //basket olduðunda yanan led pini

int mesafe; //Hcr sensörünün süresi sonucnda bulduðumuz mesafe deðeri
int sure; //Ses dalgasýnýn gidiþ geliþ süresi
int sayi = 0;


void setup() {
  //Pinleri giriþ ve çýkýþ olarak tanýmlýyoruz
 pinMode(trig,OUTPUT);
 pinMode(echo, INPUT);
 pinMode(LED, OUTPUT);

 lcd.begin(16, 2); // 16 sütun 2 satýrdan oluþuyor
void loop() {
 digitalWrite(trig, HIGH);
 delay(1);
 digitalWrite(trig, LOW);
 sure = pulseIn(echo, HIGH);
 mesafe =(sure/2)/29,1;//29.1 degeri ortamýn sýcaklýgýna göre belirlenir

 lcd.home();
 if(mesafe<=10)//Burdaki 10 cm potanýn kesilen kýsmýnýn uzunluðudur. Siz kendiniz ölçüp yazmalýsýnýz.
 {
  sayi +=1; 
  digitalWrite(LED,HIGH);
  lcd.print("ATILAN BASKET ");
  lcd.setCursor(0,1);//0.sütun 1.satýra ayarlanýyor 
  lcd.print(sayi);
  lcd.print("  OLDU");
  delay(250);
  
  }
  else
  {
    digitalWrite(LED,LOW);
    }

}
